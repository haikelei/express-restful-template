# 0419发版
## 新增用户表
```sql
CREATE TABLE `user` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `username` VARCHAR(50) COMMENT '用户昵称' DEFAULT NULL,
  `email` VARCHAR(50) COMMENT '用户邮箱' DEFAULT NULL,
  `phone` VARCHAR(20) NOT NULL COMMENT '用户手机号',
  `password` VARCHAR(100) COMMENT '密码' DEFAULT NULL,
  `avatar` VARCHAR(256) COMMENT '头像' DEFAULT NULL,
  `salt` VARCHAR(50) COMMENT 'salt值' DEFAULT NULL,
  `createTime` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updateTime` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```
## 新增验证码查询表
```sql
CREATE TABLE `verification_code` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `phone` VARCHAR(11) NOT NULL COMMENT '手机号',
  `code` VARCHAR(6) NOT NULL COMMENT '验证码',
  `createTime` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updateTime` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```
## 创建应用表
```sql
CREATE TABLE create_app_config (
  id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  phone VARCHAR(20) NOT NULL,
  status INT DEFAULT 1 COMMENT '1: 已创建, 2: 已发布,
  config JSON NOT NULL,
  createTime TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  updateTime TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```
## 创建用户最近使用APP表
```sql
CREATE TABLE user_lately_app (
  id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  phone VARCHAR(20) NOT NULL,
  appId INT NOT NULL,
  createTime TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  updateTime TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```
